Создать заказ:

var orderDetails = {
  addr: '5322 Otter Lane Middleberge FL 32068 ',
  deliveryType: 'DHL',
  deliveryCost: 35,
  payMethod: 'paypal'
};

var order1 = new Order(orderDetails)

 // Если не передать параметр с описанием свойств заказа,
// свойства будут назначены по умолчанию.
// Свойства можно изменить методом setOrderInfo()
 order1.setOrderInfo({"payMethod":"Yandex Money"});

Добавление и удаление товаров в заказ:

// Добавить товар в заказ
 order1.addItem(product1)
order1.addItem(product1)

 // Удалить товар из заказа
 order1.removeItem(product1)

 // Показать все товары в заказе
 order1.showCart()

 // Показать суммарную стоимость всех товаров в заказе
// С учетом скидки если передать в качестве аргумента объект вида
 var discount = { abs: "20 usd", pct: "10%"};
order1.getPrice(discount)

 // Удалить все товары в заказе
 order1.cleanCart()

Подтверждение заказа:

// Подтвердить заказ
// Если передан объект скидки, будет выведена информация по скидке
// и общая стоимость заказа с вычетом скидки

// Уменьшается количество товара в его свойствах
// Если количество заказанного товара превышает его количество на складе
// в заказ будет внесено максимально возможное количество. Будет выведено сообщение

// Заказу назначается порядковый номер и заказ сохраняется
// Товары из заказа удаляются

// Заказ считается завершенным
 order1.confirmOrder(discount);
